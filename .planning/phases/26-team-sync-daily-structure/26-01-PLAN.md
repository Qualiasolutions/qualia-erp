    Create `supabase/migrations/20260315_add_daily_structure.sql`:

    ```sql
    -- Daily check-ins: employees log what they did and what's blocking them
    CREATE TABLE IF NOT EXISTS daily_checkins (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      workspace_id UUID NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
      profile_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
      checkin_date DATE NOT NULL DEFAULT CURRENT_DATE,
      what_did TEXT NOT NULL,
      what_blocked TEXT,
      created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
      updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
      UNIQUE (profile_id, checkin_date)
    );

    ALTER TABLE daily_checkins ENABLE ROW LEVEL SECURITY;

    -- Employees can insert/update their own; admins can read all
    CREATE POLICY "Users can manage own checkins" ON daily_checkins
      FOR ALL USING (profile_id = auth.uid());

    CREATE POLICY "Admins can read all checkins" ON daily_checkins
      FOR SELECT USING (
        EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
      );

    -- Owner updates: admin posts notes/updates visible to specific users or all
    CREATE TABLE IF NOT EXISTS owner_updates (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      workspace_id UUID NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
      author_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
      message TEXT NOT NULL,
      target_user_id UUID REFERENCES profiles(id) ON DELETE CASCADE, -- NULL = broadcast to all
      is_pinned BOOLEAN NOT NULL DEFAULT false,
      created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    );

    ALTER TABLE owner_updates ENABLE ROW LEVEL SECURITY;

    CREATE POLICY "Admins can manage owner updates" ON owner_updates
      FOR ALL USING (
        EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
      );

    CREATE POLICY "Users can read updates targeting them or all" ON owner_updates
      FOR SELECT USING (
        target_user_id = auth.uid() OR target_user_id IS NULL
      );

    -- Acknowledgements for owner updates
    CREATE TABLE IF NOT EXISTS owner_update_reads (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      update_id UUID NOT NULL REFERENCES owner_updates(id) ON DELETE CASCADE,
      profile_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
      read_at TIMESTAMPTZ NOT NULL DEFAULT now(),
      UNIQUE (update_id, profile_id)
    );

    ALTER TABLE owner_update_reads ENABLE ROW LEVEL SECURITY;

    CREATE POLICY "Users can manage own read receipts" ON owner_update_reads
      FOR ALL USING (profile_id = auth.uid());

    CREATE POLICY "Admins can read all receipts" ON owner_update_reads
      FOR SELECT USING (
        EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
      );

    -- Task time logs: start/finish time logging per task per user
    CREATE TABLE IF NOT EXISTS task_time_logs (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      task_id UUID NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
      profile_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
      started_at TIMESTAMPTZ,
      finished_at TIMESTAMPTZ,
      time_spent_minutes INTEGER GENERATED ALWAYS AS (
        CASE
          WHEN started_at IS NOT NULL AND finished_at IS NOT NULL
          THEN EXTRACT(EPOCH FROM (finished_at - started_at)) / 60
          ELSE NULL
        END
      ) STORED,
      notes TEXT,
      created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
      UNIQUE (task_id, profile_id)
    );

    ALTER TABLE task_time_logs ENABLE ROW LEVEL SECURITY;

    CREATE POLICY "Users can manage own time logs" ON task_time_logs
      FOR ALL USING (profile_id = auth.uid());

    CREATE POLICY "Admins can read all time logs" ON task_time_logs
      FOR SELECT USING (
        EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
      );

    -- Index for fast daily_checkins lookup by date + profile
    CREATE INDEX IF NOT EXISTS idx_daily_checkins_date ON daily_checkins (checkin_date, profile_id);
    CREATE INDEX IF NOT EXISTS idx_owner_updates_workspace ON owner_updates (workspace_id, created_at DESC);
    CREATE INDEX IF NOT EXISTS idx_task_time_logs_task ON task_time_logs (task_id);
    ```
