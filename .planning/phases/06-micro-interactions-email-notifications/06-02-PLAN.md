export async function notifyPhaseSubmitted(
  projectId: string,
  projectName: string,
  phaseName: string,
  submittedByName: string
): Promise<{ success: boolean; error?: string }> {
  try {
    const resendClient = getResendClient();
    if (!resendClient) {
      console.warn('[notifyPhaseSubmitted] Resend not configured');
      return { success: true };

      .from('profiles')
      .select('email, full_name')

Fix the ghost comment bug in phase-comment-thread.tsx (lines 94-113).

Current problem: When comment creation fails, the optimistic comment persists in the UI because `useOptimistic` doesn't automatically rollback — we need to manually remove from the base state.

In handleSubmit function (line 94), update the error handling:

**Current code (lines 106-112):**

```typescript
if (!result.success) {
  setError(result.error || 'Failed to create comment');
  // Rollback: don't add to real state, optimistic will revert
} else if (result.data) {
  // Add the real comment (with server-assigned id) to persistent state
  setComments((prev) => [...prev, result.data as CommentWithProfile]);
}
```

**New code:**

```typescript
if (!result.success) {
  setError(result.error || 'Failed to create comment');
  // Rollback: Remove the optimistic comment from base state
  // The optimistic state will reflect this change automatically
  // No-op here since we never added to base state
  // The key is to NOT call setComments to add the failed comment
} else if (result.data) {
  // Add the real comment (with server-assigned id) to persistent state
  setComments((prev) => [...prev, result.data as CommentWithProfile]);
}
```

Actually, looking at the code again: the bug is that `addOptimisticComment` happens BEFORE the server action (line 95), but if the action fails, we never remove the optimistic comment from state.

The real fix is simpler — after startTransition completes, the optimistic state should revert because we're not adding to base state. But the user experience is bad because the optimistic comment stays visible during the entire async operation.

**Better fix:** Move the optimistic update INSIDE the success case only:

```typescript
startTransition(async () => {
  const result = await createPhaseComment({
    projectId,
    phaseName,
    commentText: savedText,
    isInternal: savedInternal,
  });

  if (!result.success) {
    setError(result.error || 'Failed to create comment');
    // No optimistic update was made, so nothing to roll back
  } else if (result.data) {
    // Success: add to persistent state
    setComments((prev) => [...prev, result.data as CommentWithProfile]);
  }
});

// Clear form immediately
setCommentText('');
setIsInternal(false);
```

This removes the optimistic update entirely — comments appear after server confirmation. Simpler and more reliable. The form clears immediately for good UX.

If we want to keep optimistic updates (better UX), we need to track the temp ID and remove it on failure:

```typescript
startTransition(async () => {
  addOptimisticComment(optimisticComment);

  const result = await createPhaseComment({
    projectId,
    phaseName,
    commentText: savedText,
    isInternal: savedInternal,
  });

  if (!result.success) {
    setError(result.error || 'Failed to create comment');
    // Remove the optimistic comment from base state
    // useOptimistic will reflect this automatically
    setComments((prev) => prev.filter((c) => c.id !== optimisticComment.id));
  } else if (result.data) {
    // Replace temp optimistic comment with real data
    setComments((prev) => [
      ...prev.filter((c) => c.id !== optimisticComment.id),
      result.data as CommentWithProfile,
    ]);
  }
});
```

Use the second approach (with rollback) to maintain optimistic UX while fixing the bug.
