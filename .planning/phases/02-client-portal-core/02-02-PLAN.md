Create a clean portal layout for client users:

1. Server component that wraps portal pages
2. Simple header with:
   - Qualia logo/branding on left
   - User profile dropdown on right (same as main app)
   - No admin navigation, no sidebar
3. Main content area with max-width container
4. Footer with copyright and support link (optional)

Structure:

```tsx
import { createClient } from '@/lib/supabase/server';
import { redirect } from 'next/navigation';
import { isClientRole } from '@/lib/portal-utils';

export default async function PortalLayout({ children }: { children: React.ReactNode }) {
  const supabase = await createClient();
  const {
    data: { user },
  } = await supabase.auth.getUser();

  if (!user) {
    redirect('/auth/login');
  }

  // Verify user is actually a client
  const isClient = await isClientRole(user.id);
  if (!isClient) {
    redirect('/'); // Non-clients shouldn't be here
  }

  return (
    <div className="min-h-screen bg-neutral-50">
      <header className="border-b bg-white">{/* Logo + user dropdown */}</header>
      <main className="mx-auto max-w-7xl px-4 py-8">{children}</main>
    </div>
  );
}
```
